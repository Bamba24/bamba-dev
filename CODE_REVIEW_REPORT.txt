[CRITIQUE] BUG #1: lib/posts.ts - Fonction getPostByTag retourne un seul article
────────────────────────────────────────────────────────────────────────────────
Fichier: lib/posts.ts (ligne 52-55)
Sévérité: HAUTE

Problème identifié:
    export const getPostByTag = async (tag: string, locale: string)=> {
      const posts = await getPosts(locale)
      return posts.find((post)=> post.tag === tag)  // ❌ PROBLÈME
    }

Explications:
- La fonction utilise `.find()` qui retourne UN SEUL post matching le tag
- Sur le page `/tags/[tag]/`, on affiche les ARTICLES avec ce tag (plural)
- Actuellement, seul 1 article par tag est affiché
- Les autres articles avec le même tag sont ignorés

Impact:
- Les utilisateurs voient 1 seul article par catégorie
- Les données affichées sont incorrectes

Solution:
    export const getPostByTag = async (tag: string, locale: string) => {
      const posts = await getPosts(locale);
      return posts.filter((post) => post.tag === tag);  // ✓ CORRECTION
    }

    // Hook réutilisable:
    // hooks/useDebounce.ts
    import { useState, useEffect } from 'react';
    
    export function useDebounce<T>(value: T, delay: number) {
      const [debouncedValue, setDebouncedValue] = useState<T>(value);
    
      useEffect(() => {
        const handler = setTimeout(() => {
          setDebouncedValue(value);
        }, delay);
    
        return () => clearTimeout(handler);
      }, [value, delay]);
    
      return debouncedValue;
    }

[OPT #4] Ajouter un sitemap.xml et robots.txt
────────────────────────────────────────────────────────────────────────────────

Créer app/sitemap.ts:
    import { MetadataRoute } from 'next';
    import { getPosts } from '@/lib/posts';
    
    export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
      const locales = ['fr', 'en'];
      const baseUrl = process.env.NEXT_PUBLIC_SITE_URL || 'http://localhost:3000';
      
      const entries: MetadataRoute.Sitemap = [];
    
      for (const locale of locales) {
        entries.push({
          url: `${baseUrl}/${locale}`,
          lastModified: new Date(),
          changeFrequency: 'daily',
          priority: 1,
        });
    
        const posts = await getPosts(locale);
        posts.forEach(post => {
          entries.push({
            url: `${baseUrl}/${locale}/posts/${post.slug}`,
            lastModified: new Date(post.publishedAt),
            changeFrequency: 'monthly',
            priority: 0.8,
          });
        });
      }
    
      return entries;
    }

Créer app/robots.ts:
    import { MetadataRoute } from 'next';
    
    export default function robots(): MetadataRoute.Robots {
      const baseUrl = process.env.NEXT_PUBLIC_SITE_URL || 'http://localhost:3000';
      
      return {
        rules: {
          userAgent: '*',
          allow: '/',
          disallow: '/admin',
        },
        sitemap: `${baseUrl}/sitemap.xml`,
      };
    }
